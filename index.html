// ── 방명록 ──────────────────────────────────────────── const supabaseUrl = "https://nlvkfrekpundyinhgtzo.supabase.co"; const supabaseKey = "네 publishable key"; const client = window.supabase.createClient(supabaseUrl, supabaseKey); // 방문자 UUID 생성 (사용자에게는 보이지 않음) let visitorId = localStorage.getItem("visitor_id"); if (!visitorId) { visitorId = crypto.randomUUID(); localStorage.setItem("visitor_id", visitorId); } let isSubmitting = false; async function addMessage() { if (isSubmitting) return; const msgEl = document.getElementById("msg"); const submitBtn = document.getElementById("submitBtn"); const statusMsg = document.getElementById("statusMsg"); const msg = msgEl.value.trim(); if (!msg) { statusMsg.textContent = "메시지를 입력해주세요!"; return; } isSubmitting = true; submitBtn.disabled = true; statusMsg.textContent = "등록 중..."; const { error } = await client .from("guestbook") .insert([{ nickname: "익명", message: msg, visitor_id: visitorId }]); if (error) { console.log(error); statusMsg.textContent = "등록 실패 😢 다시 시도해주세요."; isSubmitting = false; submitBtn.disabled = false; return; } msgEl.value = ""; statusMsg.textContent = "등록 완료! 🎉"; setTimeout(() => { statusMsg.textContent = ""; }, 2000); isSubmitting = false; submitBtn.disabled = false; loadMessages(); } // Enter(Shift+Enter는 줄바꿈) document.getElementById("msg").addEventListener("keydown", function(e) { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); addMessage(); } }); async function loadMessages() { const { data, error } = await client .from("guestbook") .select("*") .order("id", { ascending: false }); if (error) { console.log(error); return; } const box = document.getElementById("guestbook"); box.innerHTML = ""; (data || []).forEach(row => { const div = document.createElement("div"); div.className = "entry"; div.innerHTML = `${row.nickname || "익명"} : ${row.message}`; box.appendChild(div); }); } loadMessages();
